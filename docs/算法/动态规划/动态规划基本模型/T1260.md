### 📚 本题第一问求最长不上升子序列，第二问是贪心和二分查找：从动态规划到贪心优化


#### 一、 教学大纲与核心知识点总结

**1. 初识动态规划（寻找“最优前任”）**
* **核心思想**：避免重复计算，把小问题的答案记在表中。
* **状态定义**：$dp[i]$ 表示“以第 $i$ 个元素为结尾”的最长序列长度。
* **状态转移方程**：$dp[i] = \max(dp[j]) + 1$。
    * *讲解重点*：一定要回头看（$j < i$），并且只有当满足条件（如 $a[j] \le a[i]$）且能让队伍更长时，才去更新。


**2. 终极优化（贪心策略与二分查找）**
* **经典场景**：导弹拦截系统的第二问（最少配备多少套系统）。
* **贪心本质**：“好钢用在刀刃上”。永远选择能拦截导弹且**当前极限高度最低**的系统。
* **代码实现**：维护一个天然单调递增的 $g$ 数组。
    * 遇到新导弹 $x$：用 `lower_bound` 查找。找不到就**追加** (`push_back`)，找到了就**替换** (`*it = x`)。时间复杂度瞬间降至 $O(n \log n)$。

---


```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    vector<int> a;
    int val;
    
    // 1. 读取所有导弹的高度（应对不知道具体数量，读到文件末尾的情况）
    while (cin >> val) {
        a.push_back(val);
    }
    
    if (a.empty()) return 0;
    int n = a.size();

    // ==========================================
    // 第一问：最多能拦截的导弹数
    // 本质：求“最长不上升子序列”的长 (O(n^2) DP)
    // ==========================================
    vector<int> dp(n, 1);
    int max_intercept = 1;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (a[i] <= a[j]) { // 只要不比前面高，就能接着拦截
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
        max_intercept = max(max_intercept, dp[i]);
    }

    // ==========================================
    // 第二问：最少要配备的系统数
    // 本质：贪心策略分组，等价于求“最长上升子序列”的长 (O(n log n))
    // ==========================================
    vector<int> g; // g数组记录当前每套系统的最低拦截高度，天然是递增的
    for (int i = 0; i < n; i++) {
        // 在 g 中找第一个大于等于当前导弹高度 a[i] 的位置
        auto it = lower_bound(g.begin(), g.end(), a[i]);
        
        if (it == g.end()) {
            // 所有系统都拦截不到，必须加购新系统
            g.push_back(a[i]);
        } else {
            // 找到了能拦截且最接近的系统，更新它的极限高度
            *it = a[i]; // 你之前说的 g[it - g.begin()] = a[i] 也是完全一样的！
        }
    }

    // 3. 输出最终战报
    cout << max_intercept << endl; // 第一问答案
    cout << g.size() << endl;      // 第二问答案：g数组里有几个元素，就买了多少套系统

    return 0;
}

```
